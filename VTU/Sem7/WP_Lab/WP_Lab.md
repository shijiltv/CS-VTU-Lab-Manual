***

### Tags used:
     session_start() - Start new session or resume existing session

### Code: 
*10.php*

     <?php
     	#start session
     	session_start();
     		#check if session exists
     		if(isset($_SESSION['count']))
     			{
     				echo "Pageviews: " .$_SESSION['count'];
     				#increment the count
     				$_SESSION['count']++;
     			}
     		else
     			{
     				#if no session exists
     				$_SESSION['count']=1;
     				echo "Pageviews" .$_SESSION['count'];
     			}
     ?>
